  6. class Person{
		String name;

		public Person(String name) {
			this.name=name;
		}
		public String getName() {
			return name;
		}

		public void setName(String name) {
			this.name = name;
		}

	}

	public void change1(int x) {
		x=20;
	}

	public void change2(Integer x) {
		x=20;
	}

	public void change3(Person p) {
		p.setName("jack");
	}

	public void change4(String s) {
		s="ABC";
	}

	@Test
	public void test4() {
		int a=10;
		change1(a);
		System.out.println(a);

		Integer b=10;
		change2(b);
		System.out.println(b);

		Person p=new Person("tom");
		change3(p);
		System.out.println(p.getName());

		String s="abc";
		change4(s);
		System.out.println(s);
	}
输出是什么？

  7. class Father{
	public Father() {
		System.out.println("Father构造器");
	}
	{
		System.out.println("Father普通代码块");
	}
	static{
		System.out.println("Father Static代码块");
	}
}

class Son extends Father{
	public Son() {
		System.out.println("Son构造器");
	}
	{
		System.out.println("Son普通代码块");
	}
	static{
		System.out.println("Son Static代码块");
	}
}

@Test
	public void test1() {
		new Son();
		new Father();
		new Son();
	}
输出结果是什么？为什么？
